120  @Override
121  public void onDataConnectionStateChanged(int state) {
122  // Log.v(TAG, "Connection state changed...");
123 
124  String strState = "";
125  switch (state) {
126  case TelephonyManager.DATA_CONNECTED:
127  // send the URL on which the phone can be reached
128  String ip = "";
129  try {
130  Enumeration<NetworkInterface> nis = NetworkInterface.getNetworkInterfaces();
131  while (nis.hasMoreElements()) {
132  NetworkInterface ni = nis.nextElement();
133  Enumeration<InetAddress> iis = ni.getInetAddresses();
134  while (iis.hasMoreElements()) {
135  InetAddress ia = iis.nextElement();
136  if (ni.getDisplayName().equalsIgnoreCase("rmnet0")) {
137  ip = ia.getHostAddress();
138  }
139  }
140  }
141  } catch (Exception e) {
142  Log.e(TAG, "Error getting my own IP:", e);
143  }
144  if (ip.length() > 1) {
145  lastIp = ip;
146  }
147 
148  strState = "connected";
149 
150  break;
151  case TelephonyManager.DATA_CONNECTING:
152  strState = "connecting";
153  break;
154  case TelephonyManager.DATA_DISCONNECTED:
155  strState = "disconnected";
156  break;
157  case TelephonyManager.DATA_SUSPENDED:
158  strState = "suspended";
159  break;
160  default:
161  Log.e(TAG, "Unexpected data connection state: " + state);
162  return;
163  }
164 
165  lastDataConnectionState = strState;
166 
167  // check network type
168  ConnectivityManager connectivityManager = (ConnectivityManager) context
169  .getSystemService(Context.CONNECTIVITY_SERVICE);
170  NetworkInfo active = connectivityManager.getActiveNetworkInfo();
171  String typeName;
172  int type = -1;
173  if (active == null)
174  typeName = "none";
175  else {
176  typeName = active.getTypeName();
177  type = active.getType();
178  }
179 
180  // only send changes. Note that this method is also called when another part of the
181  // state
182  // changed.
183  if (previousConnectionType != type) {
184  previousConnectionType = type;
185 
186  // send data point immediately
187  sendDataPoint(SensorNames.CONN_TYPE, typeName, SenseDataTypes.STRING);
188  }

306  private void sendDataPoint(String sensorName, Object value, String dataType) {
307  try {
308  SensorDataPoint dataPoint = new SensorDataPoint(0);
309  if (dataType.equals(SenseDataTypes.BOOL)) {
310  dataPoint = new SensorDataPoint((Boolean) value);
311  } else if (dataType.equals(SenseDataTypes.FLOAT)) {
312  dataPoint = new SensorDataPoint((Float) value);
313  } else if (dataType.equals(SenseDataTypes.INT)) {
314  dataPoint = new SensorDataPoint((Integer) value);
315  } else if (dataType.equals(SenseDataTypes.JSON)) {
316  dataPoint = new SensorDataPoint(new JSONObject((String) value));
317  } else if (dataType.equals(SenseDataTypes.STRING)) {
318  dataPoint = new SensorDataPoint((String) value);
319  } else {
320  dataPoint = null;
321  }
322  if (dataPoint != null) {
323  notifySubscribers();
324  dataPoint.sensorName = sensorName;
325  dataPoint.sensorDescription = sensorName;
326  dataPoint.timeStamp = SNTP.getInstance().getTime();
327  sendToSubscribers(dataPoint);
328  }
329  } catch (Exception e) {
330  Log.e(TAG, "Error sending data point to subscribers of the ZephyrBioHarness");
331  }
332 
333  Intent intent = new Intent(context.getString(R.string.action_sense_new_data));
334  intent.putExtra(DataPoint.SENSOR_NAME, sensorName);
335  intent.putExtra(DataPoint.DATA_TYPE, dataType);
336  if (dataType.equals(SenseDataTypes.BOOL)) {
337  intent.putExtra(DataPoint.VALUE, (Boolean) value);
338  } else if (dataType.equals(SenseDataTypes.FLOAT)) {
339  intent.putExtra(DataPoint.VALUE, (Float) value);
340  } else if (dataType.equals(SenseDataTypes.INT)) {
341  intent.putExtra(DataPoint.VALUE, (Integer) value);
342  } else if (dataType.equals(SenseDataTypes.JSON)) {
343  intent.putExtra(DataPoint.VALUE, (String) value);
344  } else if (dataType.equals(SenseDataTypes.STRING)) {
345  intent.putExtra(DataPoint.VALUE, (String) value);
346  } else {
347  Log.w(TAG, "Error sending data point: unexpected data type! '" + dataType + "'");
348  }
349  intent.putExtra(DataPoint.TIMESTAMP, SNTP.getInstance().getTime());
350  intent.setPackage(context.getPackageName());
351  context.startService(intent);
352  }

357  @Override
358  public void startSensing(final long interval) {
359  Log.v(TAG, "Start sensing");
360 
361  SharedPreferences mainPrefs = context.getSharedPreferences(SensePrefs.MAIN_PREFS,
362  Context.MODE_PRIVATE);
363 
364  // listen to events as defined in the preferences
365  int events = 0;
366  if (mainPrefs.getBoolean(PhoneState.CALL_STATE, true)) {
367  events |= PhoneStateListener.LISTEN_CALL_STATE;
368 
369  // listen to outgoing calls
370  outgoingCallReceiver = new OutgoingCallReceiver();
371  context.registerReceiver(outgoingCallReceiver, new IntentFilter(
372  Intent.ACTION_NEW_OUTGOING_CALL));
373  }
374  if (mainPrefs.getBoolean(PhoneState.DATA_CONNECTION, true)) {
375  events |= PhoneStateListener.LISTEN_DATA_CONNECTION_STATE;
376  }
377  if (mainPrefs.getBoolean(PhoneState.UNREAD_MSG, true)) {
378  events |= PhoneStateListener.LISTEN_MESSAGE_WAITING_INDICATOR;
379  }
380  if (mainPrefs.getBoolean(PhoneState.SERVICE_STATE, true)) {
381  events |= PhoneStateListener.LISTEN_SERVICE_STATE;
382  }
383  if (mainPrefs.getBoolean(PhoneState.SIGNAL_STRENGTH, true)) {
384  events |= PhoneStateListener.LISTEN_SIGNAL_STRENGTHS;
385  }
386 
387  // start if there actually are events that we are interested in
388  if (0 != events) {
389 
390  active = true;
391  setSampleRate(interval);
392 
393  // start listening to the phone state
394  /* THIS TRIGGERS AN EXCEPTION WHEN STOPPING AND STARTING THE SENSOR */
395  telMgr.listen(phoneStateListener, events);
396 
397  alarmReceiver.start(context);
398 
399  // do the first sample immediately
400  doSample();
401 
402  } else {
403  Log.w(TAG, "Phone state sensor is started but is not registered for any events");
404  }
405 
406  }

436  private void transmitLatestState() {
437
