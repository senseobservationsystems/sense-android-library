93  // performs an in-place fft on the data in the real and imag arrays
94  // bit reversing is not necessary as the data will already be bit reversed
95  private void fft() {
96  for (int halfSize = 1; halfSize < real.length; halfSize *= 2) {
97  // float k = -(float)Math.PI/halfSize;
98  // phase shift step
99  // float phaseShiftStepR = (float)Math.cos(k);
100  // float phaseShiftStepI = (float)Math.sin(k);
101  // using lookup table
102  float phaseShiftStepR = cos(halfSize);
103  float phaseShiftStepI = sin(halfSize);
104  // current phase shift
105  float currentPhaseShiftR = 1.0f;
106  float currentPhaseShiftI = 0.0f;
107  for (int fftStep = 0; fftStep < halfSize; fftStep++) {
108  for (int i = fftStep; i < real.length; i += 2 * halfSize) {
109  int off = i + halfSize;
110  float tr = (currentPhaseShiftR * real[off]) - (currentPhaseShiftI * imag[off]);
111  float ti = (currentPhaseShiftR * imag[off]) + (currentPhaseShiftI * real[off]);
112  real[off] = real[i] - tr;
113  imag[off] = imag[i] - ti;
114  real[i] += tr;
115  imag[i] += ti;
116  }
117  float tmpR = currentPhaseShiftR;
118  currentPhaseShiftR = (tmpR * phaseShiftStepR)
119  - (currentPhaseShiftI * phaseShiftStepI);
120  currentPhaseShiftI = (tmpR * phaseShiftStepI)
121  + (currentPhaseShiftI * phaseShiftStepR);
122  }
123  }
