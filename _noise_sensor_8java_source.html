85  private float[] audioToFloat(byte[] buffer, int readBytes) {
86  float[] samples = new float[readBytes / 2];
87  int cnt = 0;
88  for (int x = 0; x < readBytes - 1; x = x + 2) {
89  double sample = 0;
90  for (int b = 0; b < BYTES_PER_SAMPLE; b++) {
91  int v = (int) buffer[x + b];
92  if (b < BYTES_PER_SAMPLE - 1 || BYTES_PER_SAMPLE == 1) {
93  v &= 0xFF;
94  }
95  sample += v << (b * 8);
96  }
97  samples[cnt++] = (float) sample;
98  }
99  return samples;
100  }

138  private double[] calculateSpectrum(float[] samples) {
139  if (samples == null || samples.length == 0)
140  return null;
141  int nrSamples = (int) Math.pow(2, (int) (Math.log(samples.length) / Math.log(2)));
142  int nrBands = (int) Math.ceil((float)DEFAULT_SAMPLE_RATE/2f / (float)FFT_BANDWITH_RESOLUTION);
143  if(nrBands > (int) Math.ceil((float)FFT_MAX_HZ/(float)FFT_BANDWITH_RESOLUTION))
144  nrBands = (int) Math.ceil((float)FFT_MAX_HZ/(float)FFT_BANDWITH_RESOLUTION);
145 
146  //remove the dc-component
147  double mean = 0;
148  for (int i = 0; i < samples.length; i++)
149  mean += samples[i];
150  mean /= samples.length;
151 
152  FFT fft = new FFT(nrSamples, DEFAULT_SAMPLE_RATE);
153  fft.forward(copyOfRange(samples, 0, nrSamples,(float)mean));
154 
155  double[] bins = new double[nrBands];
156  // computing averages does not work with small sample sizes, compute our own
157  for (int i = 0; i < nrBands; i++)
158  {
159  float avg = 0;
160  for (int j = 0; j < FFT_BANDWITH_RESOLUTION; j++) {
161  avg += fft.getFreq(i*FFT_BANDWITH_RESOLUTION+j);
162  }
163  avg /= FFT_BANDWITH_RESOLUTION;
164  bins[i] = 10.0 * Math.log10(avg);
165  }
166  return bins;
167  }
