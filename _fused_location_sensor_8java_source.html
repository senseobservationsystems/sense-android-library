137  protected LocationRequest createLocationRequest() {
138  LocationRequest locationRequest = new LocationRequest();
139  locationRequest.setInterval(getSampleRate());
140  SharedPreferences mainPrefs = context.getSharedPreferences(SensePrefs.MAIN_PREFS, Context.MODE_PRIVATE);
141  String priority = mainPrefs.getString(SensePrefs.Main.Location.FUSED_PROVIDER_PRIORITY, SensePrefs.Main.Location.FusedProviderPriority.BALANCED);
142  if(priority.equals(SensePrefs.Main.Location.FusedProviderPriority.LOW_POWER))
143  locationRequest.setPriority(LocationRequest.PRIORITY_LOW_POWER);
144  else if(priority.equals(SensePrefs.Main.Location.FusedProviderPriority.ACCURATE)){
145  locationRequest.setPriority(LocationRequest.PRIORITY_HIGH_ACCURACY);
146  locationRequest.setFastestInterval(HIGH_ACCURACY_FASTEST_INTERVAL_MILISECONDS);
147  locationRequest.setInterval(HIGH_ACCURACY_INTERVAL_MILISECONDS);
148  }
149  else{
150  locationRequest.setPriority(LocationRequest.PRIORITY_BALANCED_POWER_ACCURACY);
151  locationRequest.setFastestInterval(BALANCED_FASTEST_INTERVAL_MILISECONDS);
152  locationRequest.setInterval(BALANCED_INTERVAL_MILISECONDS);
153  }
154  return locationRequest;

157  @Override
158  public void onLocationChanged(Location fix)
159  {
160  JSONObject json = new JSONObject();
161  try {
162  json.put("latitude", fix.getLatitude());
163  json.put("longitude", fix.getLongitude());
164 
165  // always include all JSON fields, or we get problems with varying data_structure
166  json.put("accuracy", fix.hasAccuracy() ? fix.getAccuracy() : -1.0d);
167  json.put("altitude", fix.hasAltitude() ? fix.getAltitude() : -1.0);
168  json.put("speed", fix.hasSpeed() ? fix.getSpeed() : -1.0d);
169  json.put("bearing", fix.hasBearing() ? fix.getBearing() : -1.0d);
170  json.put("provider", null != fix.getProvider() ? fix.getProvider() : "unknown");
171 
172  } catch (JSONException e) {
173  Log.e(TAG, "JSONException in onLocationChanged", e);
174  return;
175  }
176 
177  // use SNTP time
178  long timestamp = SNTP.getInstance().getTime();
179  notifySubscribers();
180  SensorDataPoint dataPoint = new SensorDataPoint(json);
181  dataPoint.sensorName = SensorNames.LOCATION;
182  dataPoint.sensorDescription = SensorNames.LOCATION;
183  dataPoint.timeStamp = timestamp;
184  sendToSubscribers(dataPoint);
185 
186  // pass message to the MsgHandler
187  Intent i = new Intent(context.getString(R.string.action_sense_new_data));
188  i.putExtra(DataPoint.SENSOR_NAME, SensorNames.LOCATION);
189  i.putExtra(DataPoint.VALUE, json.toString());
190  i.putExtra(DataPoint.DATA_TYPE, SenseDataTypes.JSON);
191  i.putExtra(DataPoint.TIMESTAMP, timestamp);
192  i.setPackage(context.getPackageName());
193  context.startService(i);
194  distanceEstimator.addPoint(fix);
195  }

246  private void startAlarms() {
