350  JSONObject content, String sessionID, String requestMethod) throws IOException {
351 
352  HttpURLConnection urlConnection = null;
353  HashMap<String, String> result = new HashMap<String, String>();
354  try {
355  final boolean compress = true;
356  // open new URL connection channel.
357  URL url = new URL(urlString);
358  if ("https".equals(url.getProtocol().toLowerCase(Locale.ENGLISH))) {
359  HttpsURLConnection https = (HttpsURLConnection) url.openConnection();
360  https.setHostnameVerifier(SSLSocketFactory.ALLOW_ALL_HOSTNAME_VERIFIER);
361  urlConnection = https;
362  } else {
363  urlConnection = (HttpURLConnection) url.openConnection();
364  }
365  // some parameters
366  urlConnection.setUseCaches(false);
367  urlConnection.setInstanceFollowRedirects(false);
368  urlConnection.setRequestProperty("Accept", "application/json");
369  if(null != APP_KEY)
370  urlConnection.setRequestProperty("APPLICATION-KEY", APP_KEY);
371 
372  if(sessionID != null)
373  urlConnection.setRequestProperty("SESSION-ID", sessionID);
374 
375  if(requestMethod != null)
376  urlConnection.setRequestMethod(requestMethod);
377 
378  // send content (if available)
379  if (null != content) {
380  urlConnection.setDoOutput(true);
381  // When no charset is given in the Content-Type header "ISO-8859-1" should be
382  // assumed (see http://www.w3.org/Protocols/rfc2616/rfc2616-sec3.html#sec3.7.1).
383  // Because we're uploading UTF-8 the charset should be set to UTF-8.
384  urlConnection.setRequestProperty("Content-Type", "application/json; charset=utf-8");
385  // send content
386  DataOutputStream printout;
387  if (compress) {
388  // do not set content size
389  // use chunked transfer mode instead
390  urlConnection.setChunkedStreamingMode(0);
391  urlConnection.setRequestProperty("Content-Encoding", "gzip");
392  GZIPOutputStream zipStream = new GZIPOutputStream(
393  urlConnection.getOutputStream());
394  printout = new DataOutputStream(zipStream);
395  } else {
396  // set content size
397  // The content length should be in bytes. We cannot use string length here
398  // because that counts the number of chars while not accounting for multibyte
399  // chars
400  int contentLength = content.toString().getBytes("UTF-8").length;
401  urlConnection.setFixedLengthStreamingMode(contentLength);
402  urlConnection.setRequestProperty("Content-Length", "" + contentLength);
403  printout = new DataOutputStream(urlConnection.getOutputStream());
404  }
405  // Write the string in UTF-8 encoding
406  printout.write(content.toString().getBytes("UTF-8"));
407  printout.flush();
408  printout.close();
409  }
410 
411  // get response, or read error message
412  InputStream inputStream;
413  try {
414  inputStream = urlConnection.getInputStream();
415  } catch (IOException e) {
416  inputStream = urlConnection.getErrorStream();
417  }
418  if (null == inputStream) {
419  throw new IOException("could not get InputStream");
420  }
421  BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream), 1024);
422  String line;
423  StringBuffer responseContent = new StringBuffer();
424  while ((line = reader.readLine()) != null) {
425  responseContent.append(line);
426  responseContent.append('\r');
427  }
428  result.put(RESPONSE_CONTENT, responseContent.toString());
429  result.put(RESPONSE_CODE, Integer.toString(urlConnection.getResponseCode()));
430

432  reader.close();
433  reader = null;
