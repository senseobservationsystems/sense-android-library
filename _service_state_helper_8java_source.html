69  // icon and content text depend on the current state

87  }
88  }
89  builder.setSmallIcon(icon);
90 
91  // username will be substituted into the content text
92  final SharedPreferences authPrefs = context.getSharedPreferences(SensePrefs.AUTH_PREFS,
93  Context.MODE_PRIVATE);
94  String username = authPrefs.getString(Auth.LOGIN_USERNAME,
95  context.getString(android.R.string.unknownName));
96 
97  boolean encrypt_credential = context.getSharedPreferences(SensePrefs.MAIN_PREFS, Context.MODE_PRIVATE)
98  .getBoolean(Advanced.ENCRYPT_CREDENTIAL, false);
99 
100  if (encrypt_credential && username != context.getString(android.R.string.unknownName)) {
101  EncryptionHelper encryptor = new EncryptionHelper(context);
102  try {
103  username = encryptor.decrypt(username);
104  } catch (EncryptionHelper.EncryptionHelperException e) {
105  // Log.w(TAG, "Error decrypting username. Assume data is not encrypted");
106  }
107  }
108 
109  builder.setContentText(context.getString(contentText, username));
110  builder.setContentTitle(context.getString(R.string.stat_notify_title));
111 
112  // action to take when the notification is tapped
113  final Intent notifIntent = new Intent(context.getString(R.string.stat_notify_action));
114  notifIntent.setFlags(Intent.FLAG_ACTIVITY_NEW_TASK | Intent.FLAG_ACTIVITY_CLEAR_TOP);
115  final PendingIntent contentIntent = PendingIntent.getActivity(context, 0, notifIntent, 0);
116  builder.setContentIntent(contentIntent);
117 
118  // time of the notification
119  builder.setWhen(System.currentTimeMillis());
120  builder.setOngoing(true);
121 
122  return builder.build();
123  }
