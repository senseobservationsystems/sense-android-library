228  private void getLastKnownLocation() {
229 
230  // get the most recent location fixes
231  Location gpsFix = locMgr.getLastKnownLocation(LocationManager.GPS_PROVIDER);
232  Location nwFix = locMgr.getLastKnownLocation(LocationManager.NETWORK_PROVIDER);
233 
234  // see which provides the best recent fix
235  Location bestFix = gpsFix;
236  if (null != nwFix) {
237  // see if network provider has better recent fix
238  if (null == bestFix) {
239  // gps did not provide a recent fix
240  bestFix = nwFix;
241  } else if (nwFix.getTime() < bestFix.getTime()
242  && nwFix.getAccuracy() < bestFix.getAccuracy() + 100) {
243  // network fix is more recent and pretty accurate
244  bestFix = nwFix;
245  }
246  }
247 
248  // send best recent fix to location listener
249  if (null != bestFix) {
250  // check that the fix is not too old or from the future
251  long presentTime = SNTP.getInstance().getTime();
252  long oldTime = presentTime - 1000 * 60 * 5;
253  if (bestFix.getTime() > oldTime && bestFix.getTime() < presentTime) {
254  // use last known location as first sensor value
255  gpsListener.onLocationChanged(bestFix);
256  }
257  }
