95  public boolean requestTime(String host, int timeout) {
96  try {
97  DatagramSocket socket = new DatagramSocket();
98  socket.setSoTimeout(timeout);
99  InetAddress address = InetAddress.getByName(host);
100  byte[] buffer = new byte[NTP_PACKET_SIZE];
101  DatagramPacket request = new DatagramPacket(buffer, buffer.length, address, NTP_PORT);
102 
103  // set mode = 3 (client) and version = 3
104  // mode is in low 3 bits of first byte
105  // version is in bits 3-5 of first byte
106  buffer[0] = NTP_MODE_CLIENT | (NTP_VERSION << 3);
107 
108  // get current time and write it to the request packet
109  long requestTime = System.currentTimeMillis();
110  long requestTicks = SystemClock.elapsedRealtime();
111  writeTimeStamp(buffer, TRANSMIT_TIME_OFFSET, requestTime);
112 
113  socket.send(request);
114 
115  // read the response
116  DatagramPacket response = new DatagramPacket(buffer, buffer.length);
117  socket.receive(response);
118  long responseTicks = SystemClock.elapsedRealtime();
119  long responseTime = requestTime + (responseTicks - requestTicks);
120  socket.close();
121 
122  // extract the results
123  long originateTime = readTimeStamp(buffer, ORIGINATE_TIME_OFFSET);
124  long receiveTime = readTimeStamp(buffer, RECEIVE_TIME_OFFSET);
125  long transmitTime = readTimeStamp(buffer, TRANSMIT_TIME_OFFSET);
126  long roundTripTime = responseTicks - requestTicks - (transmitTime - receiveTime);
127  // receiveTime = originateTime + transit + skew
128  // responseTime = transmitTime + transit - skew
129  // clockOffset = ((receiveTime - originateTime) + (transmitTime - responseTime))/2
130  // = ((originateTime + transit + skew - originateTime) +
131  // (transmitTime - (transmitTime + transit - skew)))/2
132  // = ((transit + skew) + (transmitTime - transmitTime - transit + skew))/2
133  // = (transit + skew - transit + skew)/2
134  // = (2 * skew)/2 = skew
135  clockOffset = ((receiveTime - originateTime) + (transmitTime - responseTime)) / 2;
136  // if (Config.LOGD) Log.d(TAG, "round trip: " + roundTripTime + " ms");
137  // if (Config.LOGD) Log.d(TAG, "clock offset: " + clockOffset + " ms");
138 
139  // save our results - use the times on this side of the network latency
140  // (response rather than request time)
141  mNtpTime = responseTime + clockOffset;
142  mNtpTimeReference = responseTicks;
143  mRoundTripTime = roundTripTime;
144  } catch (Exception e) {
145  Log.w(TAG, "Request time failed: " + e);
146  return false;
147  }
148 
149  return true;
150  }

203  private long read32(byte[] buffer, int offset) {
204  byte b0 = buffer[offset];
205  byte b1 = buffer[offset + 1];
206  byte b2 = buffer[offset + 2];
207  byte b3 = buffer[offset + 3];
208 
209  // convert signed bytes to unsigned values
210  int i0 = ((b0 & 0x80) == 0x80 ? (b0 & 0x7F) + 0x80 : b0);
211  int i1 = ((b1 & 0x80) == 0x80 ? (b1 & 0x7F) + 0x80 : b1);
212  int i2 = ((b2 & 0x80) == 0x80 ? (b2 & 0x7F) + 0x80 : b2);
213  int i3 = ((b3 & 0x80) == 0x80 ? (b3 & 0x7F) + 0x80 : b3);
214 
215  return ((long) i0 << 24) + ((long) i1 << 16) + ((long) i2 << 8) + (long) i3;
216  }

222  private long readTimeStamp(byte[] buffer, int offset) {
223  long seconds = read32(buffer, offset);
224  long fraction = read32(buffer, offset + 4);
225  return ((seconds - OFFSET_1900_TO_1970) * 1000) + ((fraction * 1000L) / 0x100000000L);
226  }

232  private void writeTimeStamp(byte[] buffer, int offset, long time) {
233  long seconds = time / 1000L;
234  long milliseconds = time - seconds * 1000L;
235  seconds += OFFSET_1900_TO_1970;
236 
237  // write seconds in big endian format
238  buffer[offset++] = (byte) (seconds >> 24);
239  buffer[offset++] = (byte) (seconds >> 16);
240  buffer[offset++] = (byte) (seconds >> 8);
241  buffer[offset++] = (byte) (seconds >> 0);
242 
243  long fraction = milliseconds * 0x100000000L / 1000L;
244  // write fraction in big endian format
245  buffer[offset++] = (byte) (fraction >> 24);
246  buffer[offset++] = (byte) (fraction >> 16);
247  buffer[offset++] = (byte) (fraction >> 8);
248  // low order bits should be random data
249  buffer[offset++] = (byte) (Math.random() * 255.0);
250  }
