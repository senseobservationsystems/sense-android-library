128  public boolean requestTime(String host, int timeout) {
129  try {
130  DatagramSocket socket = new DatagramSocket();
131  socket.setSoTimeout(timeout);
132  InetAddress address = InetAddress.getByName(host);
133  byte[] buffer = new byte[NTP_PACKET_SIZE];
134  DatagramPacket request = new DatagramPacket(buffer, buffer.length, address, NTP_PORT);
135 
136  // set mode = 3 (client) and version = 3
137  // mode is in low 3 bits of first byte
138  // version is in bits 3-5 of first byte
139  buffer[0] = NTP_MODE_CLIENT | (NTP_VERSION << 3);
140 
141  // get current time and write it to the request packet
142  long requestTime = System.currentTimeMillis();
143  long requestTicks = SystemClock.elapsedRealtime();
144  writeTimeStamp(buffer, TRANSMIT_TIME_OFFSET, requestTime);
145 
146  socket.send(request);
147 
148  // read the response
149  DatagramPacket response = new DatagramPacket(buffer, buffer.length);
150  socket.receive(response);
151  long responseTicks = SystemClock.elapsedRealtime();
152  long responseTime = requestTime + (responseTicks - requestTicks);
153  socket.close();
154 
155  // extract the results
156  long originateTime = readTimeStamp(buffer, ORIGINATE_TIME_OFFSET);
157  long receiveTime = readTimeStamp(buffer, RECEIVE_TIME_OFFSET);
158  long transmitTime = readTimeStamp(buffer, TRANSMIT_TIME_OFFSET);
159  long roundTripTime = responseTicks - requestTicks - (transmitTime - receiveTime);
160  // receiveTime = originateTime + transit + skew
161  // responseTime = transmitTime + transit - skew
162  // clockOffset = ((receiveTime - originateTime) + (transmitTime - responseTime))/2
163  // = ((originateTime + transit + skew - originateTime) +
164  // (transmitTime - (transmitTime + transit - skew)))/2
165  // = ((transit + skew) + (transmitTime - transmitTime - transit + skew))/2
166  // = (transit + skew - transit + skew)/2
167  // = (2 * skew)/2 = skew
168  clockOffset = ((receiveTime - originateTime) + (transmitTime - responseTime)) / 2;
169  // if (Config.LOGD) Log.d(TAG, "round trip: " + roundTripTime + " ms");
170  // if (Config.LOGD) Log.d(TAG, "clock offset: " + clockOffset + " ms");
171 
172  // save our results - use the times on this side of the network latency
173  // (response rather than request time)
174  mNtpTime = responseTime + clockOffset;
175  mNtpTimeReference = responseTicks;
176  mRoundTripTime = roundTripTime;
177  } catch (Exception e) {
178  Log.w(TAG, "Request time failed: " + e);
179  return false;
180  }
181 
182  return true;
183  }

236  private long read32(byte[] buffer, int offset) {
237  byte b0 = buffer[offset];
238  byte b1 = buffer[offset + 1];
239  byte b2 = buffer[offset + 2];
240  byte b3 = buffer[offset + 3];
241 
242  // convert signed bytes to unsigned values
243  int i0 = ((b0 & 0x80) == 0x80 ? (b0 & 0x7F) + 0x80 : b0);
244  int i1 = ((b1 & 0x80) == 0x80 ? (b1 & 0x7F) + 0x80 : b1);
245  int i2 = ((b2 & 0x80) == 0x80 ? (b2 & 0x7F) + 0x80 : b2);
246  int i3 = ((b3 & 0x80) == 0x80 ? (b3 & 0x7F) + 0x80 : b3);
247 
248  return ((long) i0 << 24) + ((long) i1 << 16) + ((long) i2 << 8) + (long) i3;
249  }

255  private long readTimeStamp(byte[] buffer, int offset) {
256  long seconds = read32(buffer, offset);
257  long fraction = read32(buffer, offset + 4);
258  return ((seconds - OFFSET_1900_TO_1970) * 1000) + ((fraction * 1000L) / 0x100000000L);
259  }

265  private void writeTimeStamp(byte[] buffer, int offset, long time) {
266  long seconds = time / 1000L;
267  long milliseconds = time - seconds * 1000L;
268  seconds += OFFSET_1900_TO_1970;
269 
270  // write seconds in big endian format
271  buffer[offset++] = (byte) (seconds >> 24);
272  buffer[offset++] = (byte) (seconds >> 16);
273  buffer[offset++] = (byte) (seconds >> 8);
274  buffer[offset++] = (byte) (seconds >> 0);
275 
276  long fraction = milliseconds * 0x100000000L / 1000L;
277  // write fraction in big endian format
278  buffer[offset++] = (byte) (fraction >> 24);
279  buffer[offset++] = (byte) (fraction >> 16);
280  buffer[offset++] = (byte) (fraction >> 8);
281  // low order bits should be random data
282  buffer[offset++] = (byte) (Math.random() * 255.0);
283  }
