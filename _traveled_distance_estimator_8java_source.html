10 /***
11  * This class can be used to estimate traveled distance over a location trace, this class is
12  * especially suited to deal with sparse and/or noisy points.
13  *
14  * Implementation: (outlier filter) - first apply a median filter to remove points with a relative
15  * bad accuracy - second filter points for which we need an infeasible speed, add them to a
16  * blacklist - third filter points that are in the blacklist; this is needed since sometimes there
17  * are rogue network points that don't get filtered when the number of points is scarce (cluster) -
18  * cluster points that are within each others accuracy, the cluster point is estimated as the
19  * weighted average of the points, with weights inversely proportional to the accuracies
20  *
21  * @author Pim Nijdam <pim@sense-os.nl>
22  */

142  synchronized public void addPoint(double timestamp, double lat, double lon, double acc) {
143  PositionPoint pp = new PositionPoint(timestamp, lat, lon, acc);
144 
145  /* Median filter for accuracy outliers */
146  window_.add(pp);
147  while (window_.size() > WINDOW_SIZE) {
148  window_.remove(0);
149  }
150 
151  if (pp.acc > MEDIAN_FILTER_FACTOR * getAccuracyAtQuantile(MEDIAN_FILTER_QUANTILE))
152  return; // reject as outlier, but don't blacklist
153 
154  /* speed check */
155  if (clusterCenter_ != null) {
156  double distance = distanceBetween(clusterCenter_, pp);
157  double dt = (pp.timestamp - clusterCenter_.timestamp) / 1000;
158  if (distance > dt * MAX_SPEED + clusterCenter_.acc + pp.acc) {
159  // reject this point as outlier
160 
161  if (distance > clusterCenter_.acc + pp.acc + BLACKLIST_DISTANCE)
162  blackList(pp); // blacklist this point
163  return;
164  }
165  }
166

173  return;
174  }

242  /***
243  * update center position using the location p
244  *
245  */
246  private void updateCenter(PositionPoint p) {
247  /*
248  * center is a weighted average of the given position points, the weights are inversely
249  * proportional to the accuracy. To avoid problems with infinite weight a minimum accuracy
250  * is defined. This is an online calculation.
251  */
252 
253  // restore weighted version of the current center
254  double lat = clusterCenter_.lat * weightSum_, lon = clusterCenter_.lon * weightSum_;
255  // accuracy isn't weighted
256  double acc = clusterCenter_.acc;
257  // weight of this position is inversely proportional to the accuracy
258  double weight = 1.0 / (Math.max(p.acc, MIN_ACCURACY));
259  // accumulate weight so we can normalize the weight later
260  weightSum_ += weight;
261  // update values
262  lat += weight * p.lat;
263  lon += weight * p.lon;
264  lat /= weightSum_;
265  lon /= weightSum_;
266  // accuracy of averaged position is a bit harder to calculate, we'll
267  // just (over)estimate it with the minimum accuracy
268  acc = Math.min(acc, p.acc);
269 
270  // update center, just to be sure create a new object
271  clusterCenter_ = new PositionPoint(p.timestamp, lat, lon, acc);
272  }
