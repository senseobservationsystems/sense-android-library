172  if (whichAverage == LINAVG) {
173  int avgWidth = (int) spectrum.length / averages.length;
174  for (int i = 0; i < averages.length; i++) {
175  float avg = 0;
176  int j;
177  for (j = 0; j < avgWidth; j++) {
178  int offset = j + i * avgWidth;
179  if (offset < spectrum.length) {
180  avg += spectrum[offset];
181  } else {
182  break;
183  }
184  }
185  avg /= j + 1;
186  averages[i] = avg;
187  }
188  } else if (whichAverage == LOGAVG) {
189  for (int i = 0; i < octaves; i++) {
190  float lowFreq, hiFreq, freqStep;
191  if (i == 0) {
192  lowFreq = 0;
193  } else {
194  lowFreq = (sampleRate / 2) / (float) Math.pow(2, octaves - i);
195  }
196  hiFreq = (sampleRate / 2) / (float) Math.pow(2, octaves - i - 1);
197  freqStep = (hiFreq - lowFreq) / avgPerOctave;
198  float f = lowFreq;
199  for (int j = 0; j < avgPerOctave; j++) {
200  int offset = j + i * avgPerOctave;
201  averages[offset] = calcAvg(f, f + freqStep);
202  f += freqStep;
203  }
204  }
205  }

382  public float indexToFreq(int i) {
383  float bw = getBandWidth();
384  // special case: the width of the first bin is half that of the others.
385  // so the center frequency is a quarter of the way.
386  if (i == 0)
387  return bw * 0.25f;
388  // special case: the width of the last bin is half that of the others.
389  if (i == spectrum.length - 1) {
390  float lastBinBeginFreq = (sampleRate / 2) - (bw / 2);
391  float binHalfWidth = bw * 0.25f;
392  return lastBinBeginFreq + binHalfWidth;
393  }
394  // the center frequency of the ith band is simply i*bw
395  // because the first band is half the width of all others.
396  // treating it as if it wasn't offsets us to the middle
397  // of the band.
398  return i * bw;
399  }

407  public float getAverageCenterFrequency(int i) {
408  if (whichAverage == LINAVG) {
409  // an average represents a certain number of bands in the spectrum
410  int avgWidth = (int) spectrum.length / averages.length;
411  // the "center" bin of the average, this is fudgy.
412  int centerBinIndex = i * avgWidth + avgWidth / 2;
413  return indexToFreq(centerBinIndex);
414 
415  } else if (whichAverage == LOGAVG) {
416  // which "octave" is this index in?
417  int octave = i / avgPerOctave;
418  // which band within that octave is this?
419  int offset = i % avgPerOctave;
420  float lowFreq, hiFreq, freqStep;
421  // figure out the low frequency for this octave
422  if (octave == 0) {
423  lowFreq = 0;
424  } else {
425  lowFreq = (sampleRate / 2) / (float) Math.pow(2, octaves - octave);
426  }
427  // and the high frequency for this octave
428  hiFreq = (sampleRate / 2) / (float) Math.pow(2, octaves - octave - 1);
429  // each average band within the octave will be this big
430  freqStep = (hiFreq - lowFreq) / avgPerOctave;
431  // figure out the low frequency of the band we care about
432  float f = lowFreq + offset * freqStep;
433  // the center of the band will be the low plus half the width
434  return f + freqStep / 2;
435  }
436 
437  return 0;
438  }
