202  public byte[] getMessage(byte[] buffer, boolean useBuffer) {
203  byte[] msgBuffer = new byte[msgSize];
204  // check if we have the start from the previous run
205  if (gotStart > -1 && useBuffer) {
206  for (int i = 0; i < msgSize; i++) {
207  if (gotStart + i < bufferBuffer.size()) {
208  msgBuffer[i] = bufferBuffer.elementAt(gotStart + i);
209  } else {
210  if (gotStart + i - bufferBuffer.size() < buffer.length) {
211  msgBuffer[i] = buffer[gotStart + i - bufferBuffer.size()];
212  } else {
213  // msg to short store and bailout
214  bufferBuffer.clear();
215  for (byte element : msgBuffer) {
216  bufferBuffer.add(element);
217  gotStart = 0;
218  }
219  return new byte[0];
220  }
221  }
222  }
223  // check for the remaining part
224  int rest = msgSize - (bufferBuffer.size() - gotStart);
225  if (rest < buffer.length) {
226  bufferBuffer.clear();
227  gotStart = -1;
228  for (int i = rest; i < buffer.length; i++) {
229  bufferBuffer.add(buffer[i]);
230  if (i > 1 && buffer[i - 2] == 0x02 && buffer[i - 1] == 0x26
231  && buffer[i] == 55 && gotStart < 0) {
232  gotStart = bufferBuffer.size() - 3;
233  }
234  }
235  }
236  if (msgBuffer.length == 60) {
237  return msgBuffer;
238  } else {
239  return new byte[0];
240  }
241  } else {
242  if (!useBuffer) {
243  bufferBuffer.clear();
244  }
245 
246  // combine old buffer with new one
247  for (byte element : buffer) {
248  bufferBuffer.add(element);
249  }
250  // find the start in the current buffer
251  for (int i = 0; i < bufferBuffer.size(); i++) {
252  // got the start copy the rest
253  if (i + 2 < bufferBuffer.size() && bufferBuffer.get(i) == 0x02
254  && bufferBuffer.get(i + 1) == 0x26 && bufferBuffer.get(i + 2) == 55) {
255  // can it hold a message
256  if (bufferBuffer.size() - i >= 60) {
257  Vector<Byte> tempBuffer = new Vector<Byte>();
258  for (int j = 0; j + i < bufferBuffer.size(); j++) {
259  if (j < 60) {
260  msgBuffer[j] = bufferBuffer.get(i + j);
261  } else { // copy the remaining part in the temp buffer
262  if (i + j + 2 < bufferBuffer.size()
263  && bufferBuffer.get(i + j) == 0x02
264  && bufferBuffer.get(i + j + 1) == 0x26
265  && bufferBuffer.get(i + j + 2) == 55 && gotStart < 0) {
266  gotStart = 60 - j;
267  }
268  tempBuffer.add(bufferBuffer.get(i + j));
269  }
270  }
271  // copy the tempbuffer in the main buffer
272  bufferBuffer.clear();
273  for (int j = 0; j < tempBuffer.size(); j++) {
274  bufferBuffer.addElement(tempBuffer.get(j));
275  }
276  return msgBuffer;
277  } else {
278  return new byte[0];
279  }
280  }
281  }
282  }
283  return new byte[0];
284  }
