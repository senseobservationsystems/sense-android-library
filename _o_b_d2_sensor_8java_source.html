185  public void execute(){
186  try{
187  //determine the endtime of this execution
188  long endtime = System.currentTimeMillis() + maximumexecutetime;
189 
190  //send the command once
191  sendRequest();
192 
193  while(connected && System.currentTimeMillis() < endtime){
194  //read the response
195  //TODO: if this does not work all the time, have a thorough look at readUpToPrompt from ELMBT.java in com.gtosoft.libvoyager.android;
196  receiveResponse();
197  if(this.hasValidResponse()){
198  //trim the data
199  data.trim();
200  return;
201  }
202  try{
203  Thread.sleep(executioninterval);
204  }
205  catch(InterruptedException e){
206  Log.e(TAG, "trouble sleeping while executing command ",e);
207  }
208  }
209  if(!this.hasValidResponse()){
210  Log.e(TAG, "command " + this.getCommand() + " did not get a valid response. Instead got: " +this.getData());
211  }
212  return;
213  }
214  catch (Exception e) {
