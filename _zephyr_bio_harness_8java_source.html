180  // received general data
181  if (buffer[0] == 0x02 && buffer[1] == 0x20 && buffer[2] == 53) {
182  // check that payload is not completely empty
183  boolean hasPayload = false;
184  for (int i = 12; i < 58; i++) {
185  if (buffer[i] != 0) {
186  hasPayload = true;
187  break;
188  }
189  }
190  if (false == hasPayload) {
191  Log.w(TAG, "No sensor data payload received");
192  // return true because the message type was correct, only the data is wrong
193  return true;
194  }
195 
196  // send acceleration data in m/s^2
197  if (prefs.getBoolean(External.ZephyrBioHarness.ACC, true)) {
198  float g = 9.80665f;
199  JSONObject json = new JSONObject();
200  Short xmin = (short) (buffer[32] | buffer[33] << 8);
201  Short xmax = (short) (buffer[34] | buffer[35] << 8);
202  float x = (xmin.floatValue() + xmax.floatValue()) / 200.0f * g;
203  Short ymin = (short) (buffer[36] | buffer[37] << 8);
204  Short ymax = (short) (buffer[38] | buffer[39] << 8);
205  float y = (ymin.floatValue() + ymax.floatValue()) / 200.0f * g;
206  Short zmin = (short) (buffer[40] | buffer[41] << 8);
207  Short zmax = (short) (buffer[42] | buffer[43] << 8);
208  float z = (zmin.floatValue() + zmax.floatValue()) / 200.0f * g;
209  // Log.v(TAG, "x:" + x + " y:" + y + " z:" + z);
210 
211  json.put("x-axis", x);
212  json.put("y-axis", y);
213  json.put("z-axis", z);
214  sendDataPoint(SensorNames.ACCELEROMETER, "BioHarness " + deviceType,
215  json.toString(), SenseDataTypes.JSON);
216  }
217 
218  // send heart rate
219  if (prefs.getBoolean(External.ZephyrBioHarness.HEART_RATE, true)) {
220  Short heartRate = (short) (buffer[12] | buffer[13] << 8);
221  // if(heartRate < (short)0)
