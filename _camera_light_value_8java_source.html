42  private float calculateLightValue(byte[] data, int width, int height) {
43  try {
44  float lux = 0;
45  // Data is YCrCb NV21 encoding
46  // the first height*width are Y: luminance values
47 
48  // calculate the average luminance
49  for (int i = 0; i < height * width; i++) {
50  // conversion to java unsigned byte
51  if (data[i] < 0)
52  lux += (data[i] + 256);
53  else
54  lux += data[i];
55  }
56  return lux /= ((float) width * height);
57  } catch (Exception e) {
